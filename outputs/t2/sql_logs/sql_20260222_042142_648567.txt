        event_id,
        worker_id,
        tool,
        t_start,
        t_end,
        LAG(tool, 1) OVER (PARTITION BY worker_id ORDER BY t_start) AS previous_tool
    FROM
        Events
    WHERE
        worker_id = 't2'
),
ToolBlocks AS (
    SELECT
        event_id,
        worker_id,
        tool,
        t_start,
        t_end,
        -- Generate a new block_id whenever the tool changes (including changes to/from NULL)
        SUM(CASE WHEN (tool IS NOT previous_tool) THEN 1 ELSE 0 END) OVER (PARTITION BY worker_id ORDER BY t_start) AS block_id
    FROM
        RankedEvents
)
SELECT
    tool,
    (MAX(t_end) - MIN(t_start)) AS duration_seconds,
    MIN(t_start) AS t_start,

FROM

GROUP BY

ORDER BY
